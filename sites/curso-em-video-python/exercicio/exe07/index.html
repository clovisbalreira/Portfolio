Operadores aritméticos

n1 = 5 n2 = 2 print('Adição {} + {} = {}'.format(n1,n2,n1+n2)) print('Subtração {} - {} = {}'.format(n1,n2,n1-n2)) print('Multiplicação {} * {} = {}'.format(n1,n2,n1*n2)) print('Divisão {} / {} = {}'.format(n1,n2,n1/n2)) print('Potência {} ** {} = {}'.format(n1,n2,n1**n2)) print('Divisão interira {} // {} = {}'.format(n1,n2,n1//n2)) print('Resto da divisão {} % {} = {}'.format(n1,n2,n1%n2)) print('Ordem de precedência') print('1º - ()') print('2º - **') print('3º - * / // %') print('4º - + -') print('Exemplo:') print('5 + 3 * 2 = {:.3f}'.format(5+3*2)) print('3 * 5 + 4 ** 2 = {}'.format(3*5+4**2)) print('3 * ( 5 + 4 ) ** 2 = {}'.format(3*(5+4)**2))
